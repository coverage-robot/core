SELECT
  COALESCE(
    ARRAY_AGG(uploadId),
    []
  ) as successfulUploads,
  COALESCE(
    ARRAY_AGG(
      STRING(ingestTime)
    ),
    []
  ) as successfulIngestTimes,
  COALESCE(
    ARRAY_AGG(
      STRUCT(tag as name, @COMMIT as commit)
    ),
    []
  ) as successfulTags
FROM
  `mock-table`
WHERE
  provider = @PROVIDER
  AND owner = @OWNER
  AND repository = @REPOSITORY
  AND commit = @COMMIT
